Show the spinner and hide the input content initially document.querySelector('.spinner-container').classList.remove('hidden') document.querySelector('.input-content').classList.add('hidden') # Install Pillow console.log("Loading Pillow...") await micropip.install("pillow") console.log("Loaded Pillow.") from PIL import Image # After loading Pillow, hide the spinner and show the input content document.querySelector('.spinner-container').classList.add('hidden') document.querySelector('.input-content').classList.remove('hidden') # Initializing the lists thumbnail_list = document.getElementById("thumbnail-list") converted_list = document.getElementById("converted-list") items = [] converted_items = [] async def add_file(event): files = event.target.files for i in range(files.length): file = files.item(i) blob_url = URL.createObjectURL(file) item_data = {"type": "file", "data": file, "src": blob_url, "name": file.name} items.append(item_data) display_thumbnail(blob_url, file.name, thumbnail_list, item_data) async def add_directory(event): files = event.target.files for i in range(files.length): file = files.item(i) if file.type.startswith("image/"): blob_url = URL.createObjectURL(file) item_data = {"type": "file", "data": file, "src": blob_url, "name": file.name} items.append(item_data) display_thumbnail(blob_url, file.name, thumbnail_list, item_data) async def add_url(event): url = document.getElementById("input-url").value.strip() if url: try: response = await pyfetch(url) byte_data = await response.bytes() item_data = {"type": "url", "data": byte_data, "src": url, "name": url.split("/")[-1]} items.append(item_data) display_thumbnail(url, url.split("/")[-1], thumbnail_list, item_data) document.getElementById("input-url").value = "" except Exception as e: console.log(f"Failed to fetch image from URL: {url}, error: {e}") def display_thumbnail(src, name, container, item_data, is_converted=False): thumbnail_item = document.createElement("div") thumbnail_item.className = "flex items-center gap-2 mb-2" # Remove Button remove_btn = document.createElement("button") remove_btn.innerHTML = "✕" # Cross symbol remove_btn.className = "text-red-500 hover:text-red-700 text-xl focus:outline-none" remove_btn.onclick = create_proxy(lambda e: remove_image(item_data, container, thumbnail_item)) # Image Element img = document.createElement("img") img.src = src img.alt = name img.className = "w-12 h-12 object-cover rounded border" # Name Span text = document.createElement("span") text.textContent = name text.className = "flex-1 truncate" # Append Elements in order thumbnail_item.appendChild(remove_btn) thumbnail_item.appendChild(img) thumbnail_item.appendChild(text) if is_converted: # Save Icon save_link = document.createElement("a") save_link.href = src save_link.download = name save_link.innerHTML = "💾" # Floppy disk symbol save_link.className = "text-green-500 hover:text-green-700 text-xl ml-2" thumbnail_item.appendChild(save_link) converted_items.append(item_data) container.appendChild(thumbnail_item) def remove_image(item_data, container, thumbnail_item): if container.id == "thumbnail-list": if item_data in items: items.remove(item_data) elif container.id == "converted-list": if item_data in converted_items: converted_items.remove(item_data) container.removeChild(thumbnail_item) async def convert_images(event): if not items: console.log("No images to convert.") return converted_list.innerHTML = "" converted_items.clear() selected_format = document.getElementById("format-select").value # Show spinner during conversion document.querySelector('.spinner-container').classList.remove('hidden') document.querySelector('.input-content').classList.add('hidden') for item in items: if item["type"] == "file": array_buffer = await item["data"].arrayBuffer() byte_data = bytes(array_buffer.to_py()) else: byte_data = item["data"] img = Image.open(io.BytesIO(byte_data)) if img.mode == "P": img = img.convert("RGBA") output_io = io.BytesIO() if selected_format == "JPEG": img = img.convert("RGB") img.save(output_io, format=selected_format) output_io.seek(0) original_name = item["name"].rsplit(".", 1)[0] download_name = f"{original_name}.{selected_format.lower()}" converted_blob = Blob.new([Uint8Array.from_(output_io.read())], {"type": f"image/{selected_format.lower()}"}) blob_url = URL.createObjectURL(converted_blob) # Create a new item_data for the converted item converted_item_data = {"name": download_name, "data": output_io.getvalue()} display_thumbnail(blob_url, download_name, converted_list, converted_item_data, is_converted=True) # Hide spinner and show input content document.querySelector('.spinner-container').classList.add('hidden') document.querySelector('.input-content').classList.remove('hidden') document.getElementById("download-all-button").classList.remove('hidden') def download_all(event): zip_io = io.BytesIO() with zipfile.ZipFile(zip_io, mode="w", compression=zipfile.ZIP_DEFLATED) as zip_file: filename_counts = {} for item in converted_items: base_name = item['name'] name, ext = base_name.rsplit('.', 1) count = filename_counts.get(base_name, 0) if count > 0: new_name = f"{name}({count}).{ext}" else: new_name = base_name filename_counts[base_name] = count + 1 zip_file.writestr(new_name, item['data']) zip_blob = Blob.new([Uint8Array.from_(zip_io.getvalue())], {"type": "application/zip"}) zip_url = URL.createObjectURL(zip_blob) download_link = document.createElement("a") download_link.href = zip_url download_link.download = "converted_images.zip" download_link.click() # Event handlers document.getElementById("input-file").addEventListener("change", create_proxy(add_file)) document.getElementById("directory-file").addEventListener("change", create_proxy(add_directory)) document.getElementById("add-url-button").addEventListener("click", create_proxy(add_url)) document.getElementById("convert-button").addEventListener("click", create_proxy(convert_images)) document.getElementById("download-all-button").addEventListener("click", create_proxy(download_all)) # Handle Enter key press in URL input async def input_url_keydown(event): if event.key == 'Enter': await add_url(event) document.getElementById("input-url").addEventListener("keydown", create_proxy(input_url_keydown)) # Schedule the 'main' coroutine asyncio.ensure_future(main())
